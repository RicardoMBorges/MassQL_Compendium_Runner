#MonoChloride Compounds
QUERY scaninfo(MS2DATA) WHERE 
MS1MZ=X:INTENSITYMATCH=Y:INTENSITYMATCHREFERENCE 
AND MS1MZ=X+2:INTENSITYMATCH=Y*(0.346):INTENSITYMATCHPERCENT=15
AND MS2PREC=X:TOLERANCEMZ=0.02

#DiChloride Compounds
QUERY scaninfo(MS2DATA) WHERE 
MS1MZ=X:INTENSITYMATCH=Y:INTENSITYMATCHREFERENCE 
AND MS1MZ=X+2:INTENSITYMATCH=Y*(0.67):INTENSITYMATCHPERCENT=15 
AND MS1MZ=X+4:INTENSITYMATCH=Y*(0.119):INTENSITYMATCHPERCENT=15
AND MS2PREC=X:TOLERANCEMZ=0.02:INTENSITYPERCENT=10

#TriChloride Compounds
QUERY scaninfo(MS2DATA) WHERE 
MS1MZ=X:INTENSITYMATCH=Y:INTENSITYMATCHREFERENCE 
AND MS1MZ=X+2:INTENSITYMATCH=Y*(1):INTENSITYMATCHPERCENT=15 
AND MS1MZ=X+4:INTENSITYMATCH=Y*(0.336):INTENSITYMATCHPERCENT=15
AND MS2PREC=X:TOLERANCEMZ=0.02

#TetraChloride Compounds
QUERY scaninfo(MS2DATA) WHERE 
MS1MZ=X:INTENSITYMATCH=Y:INTENSITYMATCHREFERENCE 
AND MS1MZ=X+2:INTENSITYMATCH=Y*(1.31):INTENSITYMATCHPERCENT=15 
AND MS1MZ=X+4:INTENSITYMATCH=Y*(0.658):INTENSITYMATCHPERCENT=15
AND MS1MZ=X+6:INTENSITYMATCH=Y*(0.15):INTENSITYMATCHPERCENT=15
AND MS2PREC=X:TOLERANCEMZ=0.02

#MonoBrominated Compounds
QUERY scaninfo(MS2DATA) WHERE 
MS1MZ=X:INTENSITYMATCH=Y:INTENSITYMATCHREFERENCE 
AND MS1MZ=X+2:INTENSITYMATCH=Y*(0.997):INTENSITYMATCHPERCENT=15
AND MS2PREC=X:TOLERANCEMZ=0.02

#DiBrominated Compounds
QUERY scaninfo(MS2DATA) WHERE 
MS1MZ=X:INTENSITYMATCH=Y:INTENSITYMATCHREFERENCE 
AND MS1MZ=X+2:INTENSITYMATCH=Y*(2):INTENSITYMATCHPERCENT=20
AND MS1MZ=X+4:INTENSITYMATCH=Y*(1):INTENSITYMATCHPERCENT=20
AND MS2PREC=X:TOLERANCEMZ=0.02

#TriBrominated Compounds
QUERY scaninfo(MS2DATA) WHERE 
MS1MZ=X:INTENSITYMATCH=Y:INTENSITYMATCHREFERENCE 
AND MS1MZ=X+2:INTENSITYMATCH=Y*(3):INTENSITYMATCHPERCENT=20
AND MS1MZ=X+4:INTENSITYMATCH=Y*(3):INTENSITYMATCHPERCENT=20
AND MS1MZ=X+6:INTENSITYMATCH=Y*(1):INTENSITYMATCHPERCENT=20
AND MS2PREC=X:TOLERANCEMZ=0.02

#TetraBrominated Compounds
QUERY scaninfo(MS2DATA) WHERE 
MS1MZ=X:INTENSITYMATCH=Y:INTENSITYMATCHREFERENCE 
AND MS1MZ=X+2:INTENSITYMATCH=Y*(4):INTENSITYMATCHPERCENT=20
AND MS1MZ=X+4:INTENSITYMATCH=Y*(6):INTENSITYMATCHPERCENT=20
AND MS1MZ=X+6:INTENSITYMATCH=Y*(4):INTENSITYMATCHPERCENT=20
AND MS2PREC=X:TOLERANCEMZ=0.02
